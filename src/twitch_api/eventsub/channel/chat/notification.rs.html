69/// Information about the user that triggered this notification

72pub enum Chatter {

138impl<'de> serde::Deserialize<'de> for Chatter {
139    fn deserialize<D>(deserializer: D) -> Result<Self, D::Error>
140    where D: serde::Deserializer<'de> {
141        #[derive(Deserialize)]
142        struct InnerChatter {
143            chatter_user_id: Option<types::UserId>,
144            chatter_user_name: Option<types::DisplayName>,
145            chatter_user_login: Option<types::UserName>,
146            color: Option<types::HexColor>,
147            chatter_is_anonymous: bool,
148        }
149
150        let chatter = InnerChatter::deserialize(deserializer)?;
151        if chatter.chatter_is_anonymous {
152            #[cfg(feature = "tracing")]
153            if let Some(c) = chatter.color {
154                if c.as_str() != "" {
155                    tracing::error!("got an anonymous user with color set to {c}");
156                }
157            }
158            Ok(Chatter::Anonymous)
159        } else {
160            Ok(Chatter::Chatter {
161                chatter_user_id: chatter
162                    .chatter_user_id
163                    .ok_or_else(|| serde::de::Error::missing_field("chatter_user_id"))?,
164                chatter_user_name: chatter
165                    .chatter_user_name
166                    .ok_or_else(|| serde::de::Error::missing_field("chatter_user_name"))?,
167                chatter_user_login: chatter
168                    .chatter_user_login
169                    .ok_or_else(|| serde::de::Error::missing_field("chatter_user_login"))?,
170                color: chatter
171                    .color
172                    .ok_or_else(|| serde::de::Error::missing_field("color"))?,
173            })
174        }
175    }
176}
